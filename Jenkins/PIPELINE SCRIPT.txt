pipeline {
    agent {label "projectb"}

    stages {
        stage('step 1: Git clone') {
            steps {
                git branch: 'main', url: 'https://github.com/Sachinnayak0712/barista_cafe.git'
            }
        }
        stage('step 2 : Docker build') {
            steps {
                sh 'docker build -t barista_cafe .'
            }
        }
        stage('step 3 : docker tag') {
            steps {
                sh 'docker tag barista_cafe sachinnayak0712/barista_cafe:1.0'
            
            }
        }
        stage('step 4 : docker run') {
            steps {
                sh 'docker stop barista_cafe_container || true'
                sh 'docker rm barista_cafe_container || true'
                sh 'docker run -d --name barista_cafe_container -p 80:80 sachinnayak0712/barista_cafe:1.0'

            }
        }
        stage('step 5 : docker login') {
            steps {
                sh 'docker login -u sachinnayak0712 -p Sachinnayak@0712'
            }
        }
        stage('step 5 : docker push') {
            steps {
                sh 'docker push sachinnayak0712/barista_cafe:1.0'
            }
        }
        
    }
}
